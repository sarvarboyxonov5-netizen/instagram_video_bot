@bot.message_handler(func=lambda message: 'instagram.com' in message.text)
def download_instagram_video(message):
    url = message.text.strip()
    msg = bot.send_message(message.chat.id, "â³ Videoni yuklab olayapman, biroz kuting...")
    try:
        # Instagram downloader API
        api_url = f"https://insta-saver-apis.vercel.app/api/instagram?url={url}"
        response = requests.get(api_url, timeout=30)

        if response.status_code == 200:
            data = response.json()
            if 'video_url' in data:
                video_url = data['video_url']
                video_file = requests.get(video_url)
                with open("video.mp4", "wb") as f:
                    f.write(video_file.content)
                bot.delete_message(message.chat.id, msg.message_id)
                bot.send_video(message.chat.id, open("video.mp4", "rb"))
                os.remove("video.mp4")
            else:
                bot.edit_message_text("âš ï¸ Kechirasiz, videoni yuklab boâ€˜lmadi. Linkni tekshirib qayta urinib koâ€˜ring.", 
                                      chat_id=message.chat.id, message_id=msg.message_id)
        else:
            bot.edit_message_text("âŒ Server bilan aloqa xatosi. Keyinroq urinib koâ€˜ring.", 
                                  chat_id=message.chat.id, message_id=msg.message_id)
    except Exception as e:
